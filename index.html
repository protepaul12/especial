import tkinter as tk from tkinter import messagebox def mostrar_mensaje(): mensaje = """Querida Sayu, En cada sonrisa tuya hay luz, en cada palabra, calidez, y en cada momento contigo, un regalo invaluable. Que este pequeño detalle refleje cuánto te apreciamos. Con amor, [Tu nombre] """ messagebox.showinfo("Mensaje para Sayu", mensaje) # Crear la ventana principal ventana = tk.Tk() ventana.title("Un regalo especial para Sayu") ventana.geometry("400x300") # Crear un título titulo = tk.Label(ventana, text="Para la increíble Sayu", font=("Arial", 18, "bold"), fg="darkred") titulo.pack(pady=20) # Botón para mostrar el mensaje boton = tk.Button(ventana, text="Haz clic para ver tu regalo 💝", font=("Arial", 14), command=mostrar_mensaje, bg="pink") boton.pack(pady=50) ventana.mainloop()
